File Handling Example

void saveFlightData(Flight flights[], int count) {
    ofstream file("flights.txt");
    
    for(int i = 0; i < count; i++) {
        file << flights[i].flightNumber << " "
             << flights[i].destination << " "
             << flights[i].departureTime << " "
             << flights[i].arrivalTime << " "
             << flights[i].totalSeats << " "
             << flights[i].availableSeats << " "
             << flights[i].fare << " "
             << flights[i].status << endl;
    }
    
    file.close();
}

void loadFlightData(Flight flights[], int& count) {
    ifstream file("flights.txt");
    count = 0;
    
    while(file >> flights[count].flightNumber 
               >> flights[count].destination
               >> flights[count].departureTime
               >> flights[count].arrivalTime
               >> flights[count].totalSeats
               >> flights[count].availableSeats
               >> flights[count].fare
               >> flights[count].status) {
        count++;
    }
    
    file.close();
}

Menu System Example

void passengerMenu(Passenger* p) {
    int choice;
    
    do {
        cout << "\n=== PASSENGER MENU ===" << endl;
        cout << "1. View Available Flights" << endl;
        cout << "2. Book a Flight" << endl;
        cout << "3. Cancel Reservation" << endl;
        cout << "4. View My Bookings" << endl;
        cout << "5. Logout" << endl;
        cout << "Enter choice: ";
        cin >> choice;
        
        switch(choice) {
            case 1:
                viewAvailableFlights();
                break;
            case 2:
                bookFlight(p);
                break;
            case 3:
                cancelReservation(p);
                break;
            case 4:
                generatePassengerReport(p->id);
                break;
            case 5:
                cout << "Logging out..." << endl;
                break;
            default:
                cout << "Invalid choice!" << endl;
        }
    } while(choice != 5);
}
